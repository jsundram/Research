\section{Introduction}

\begin{defn} The \emph{maximum independent cover number }of a graph $G$
	is the maximum $\mic(G)$ of $\size{I, V(G) \setminus I}$ over all independent sets $I$
	of $G$. 
\end{defn}

\begin{KernelMagic} Let $G = (V,E)$ be a nonempty graph and $\func{f}{V}{\IN}$ with $f(v)\le d_{G}(v)+1$
for all $v\in V$. If there is an independent $A\subseteq V$ such that 
\[
\size{A,V}\ge\sum_{v\in V}d_{G}(v)+1-f(v),
\]
then $G$ has a nonempty induced subgraph $H$ that is online $(rf_{H}:r)$-choosable
where $f_{H}(v)\DefinedAs f(v)+d_{H}(v)-d_{G}(v)$ for $v\in V(H)$. 
\end{KernelMagic}
